POST /api/job_invocations
Create a job invocation

Examples

POST /api/job_invocations
{
  "job_invocation": {
    "job_category": "Job name 237",
    "name": "RandomName",
    "job_template_id": 1007982151,
    "targeting_type": "static_query",
    "search_query": "foobar",
    "recurrence": {
      "cron_line": "foo"
    }
  }
}
500
{
  "error": {
    "message": "Validation failed: Triggering: Cronline foo is not valid format of cron line"
  }
}
