DELETE /api/hosts/bulk
Delete multiple hosts

Examples

DELETE /api/hosts/bulk
{
  "search": "id ^ (980190962,298486374)",
  "hosts_bulk_action": {
    "search": "id ^ (980190962,298486374)"
  }
}
200
{
  "message": "Deleted 2 hosts"
}
